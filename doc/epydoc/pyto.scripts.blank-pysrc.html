  1  #!/usr/bin/env python 
  2  """ 
  3  Blanks (sets to a given value) those voxels of a tomogram that correspond to 
  4  given segments.  
  5   
  6  This script may be placed anywhere in the directory tree. 
  7   
  8  $Id: blank.py 21 2007-12-23 11:38:03Z vladan $ 
  9  """ 
 10   
 11  import os 
 12  import numpy 
 13  import pyto 
 14   
 15   
 16  ############################################################## 
 17  # 
 18  # Parameters 
 19  # 
 20  ############################################################## 
 21   
 22  ############################################################ 
 23  # 
 24  # Image (grayscale) file. If it isn't in em or mrc format, format related 
 25  # variables (shape, data type, byte order and array order) need to be given 
 26  # (see labels file section below). 
 27  # 
 28   
 29  # name of the image file 
 30  imageFileName = "and-1-6.mrc" 
 31   
 32  ##################################################### 
 33  # 
 34  # Labels file. If the file is in em or mrc format shape, data type, byte order 
 35  # and array order are not needed (should be set to None). If these variables 
 36  # are specified they will override the values given in em or mrc headers. 
 37  # 
 38   
 39  # name of the labels file (file containing segmented volume)  
 40  labelsFileName = "../viz/and-1-6_labels_sv.raw" 
 41   
 42  # labels file dimensions 
 43  labelsShape = (260, 260, 100) 
 44   
 45  # labels file data type (e.g. 'int8', 'int16', 'int32', 'float16', 'float64')  
 46  labelsDataType = 'int8' 
 47   
 48  # labels file byteOrder ('<' for little-endian, '>' for big-endian) 
 49  labelsByteOrder = '<' 
 50   
 51  # labels file array order ('FORTRAN' for x-axis fastest, 'C' for z-axis fastest) 
 52  labelsArrayOrder = 'FORTRAN' 
 53   
 54  # offset of labels in respect to the image (None means 0-offset) 
 55  labelsOffset = None 
 56   
 57   
 58  ######################################### 
 59  # 
 60  # Calculation related parameters 
 61  # 
 62   
 63  # ids of all labels that need to be blanked (removed)  
 64  #vesicleIds = range(2,64)  # range of ids 
 65  ids = [2,3,4]       # individual ids 
 66  #ids = range(2,5)   # range of ids, same as above 
 67   
 68  # value to which the blanked elements are set 
 69  value = -1. 
 70   
 71  ########################################################### 
 72  # 
 73  # Output (blanked) file. The output file name is formed as: 
 74  # 
 75  #   labels_directory + labels_base_name + <connSuffix> + labels_extension 
 76  # 
 77   
 78  # out file name suffix 
 79  blankSuffix = '_blank-test' 
 80   
 81   
 82  ################################################################ 
 83  # 
 84  # Work 
 85  # 
 86  ################################################################ 
 87   
 88  ################################################ 
 89  # 
 90  # Read files 
 91  # 
 92   
 93  # read image file 
 94  imageFile = pyto.io.ImageIO() 
 95  imageFile.read(file=imageFileName) 
 96   
 97  # read labels file 
 98  labels = pyto.io.ImageIO() 
 99  labels.read(file=labelsFileName, byteOrder=labelsByteOrder, dataType=labelsDataType, 
100              arrayOrder=labelsArrayOrder, shape=labelsShape) 
101   
102  ################################################# 
103  # 
104  # Calculations 
105  # 
106   
107  blankData = imageFile.data.copy() 
108  for id in ids: 
109      blankData[labels.data==id] = value 
110   
111  ################################################## 
112  # 
113  # Write output file 
114  # 
115   
116  # extract parts from the imageFileName and make blank image name 
117  (dir, base) = os.path.split(imageFileName) 
118  (root, ext) = os.path.splitext(base) 
119  blankFileName = os.path.join(dir, root+blankSuffix+ext) 
120   
121  # write the blank file with the same header as in imageFile 
122  imageFile.write(file=blankFileName, data=blankData)
